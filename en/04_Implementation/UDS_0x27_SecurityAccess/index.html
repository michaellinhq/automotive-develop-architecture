/**

 * @param[in]  SeedLen - Expected seed length

Std_ReturnType Dcm_GetSeed(
    uint8 SecurityAccessType,
    uint8 SeedLen,
    uint8* Seed,
    Dcm_OpStatusType OpStatus)
{

        retVal = E_NOT_OK;
    }
    else if (SeedLen > DCM_SECURITY_SEED_SIZE)

        retVal = E_NOT_OK;
    }
    else

        {
            case DCM_INITIAL:
                /* è°ƒç”¨ Csm ç”Ÿæˆéšæœºæ•° */

                    CSM_JOB_ID_RANDOM_SEED,
